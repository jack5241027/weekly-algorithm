# 133. Clone Graph

## [題目連結](https://leetcode.com/problems/clone-graph/)

## Ｃode

```javascript
/**
 * // Definition for a Node.
 * function Node(val,neighbors) {
 *    this.val = val;
 *    this.neighbors = neighbors;
 * };
 */
const getNodes = (node) => {
  const que = [];
  const map = new Map();

  que.push(node);
  map.set(node, node);

  while (que.length) {
    const head = que.shift();
    const nbs = head.neighbors;
    for (let i = 0; i < nbs.length; i++) {
      const nb = nbs[i];
      if (!map.has(nb)) {
        que.push(nb);
        map.set(nb, nb);
      }
    }
  }

  return map;
}

var cloneGraph = function(node) {
  if (node === null) return node;
  const nodes = getNodes(node);
  const mapping = new Map();

  nodes.forEach((head) => {
    mapping.set(head, new Node(head.val, []));
  });

  nodes.forEach((head) => {
    const newNode = mapping.get(head);
    for (let i = 0; i < head.neighbors.length; i++) {
      const newNb = mapping.get(head.neighbors[i]);
      newNode.neighbors.push(newNb);
    }
  });

  return mapping.get(node);
};
```
